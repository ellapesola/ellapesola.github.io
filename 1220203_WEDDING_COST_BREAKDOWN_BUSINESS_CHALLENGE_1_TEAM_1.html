#IMPORTING NUMPY PACKAGES
import numpy as np

department_expense['percentage'] = department_expense['department_expense'] / department_expense['department_expense'].sum() * 100

#FUNCTION TO GENERATE SECTION HEADERS
def section_header(header_text):
    return f"""
----------------------------------------------------------------------------------------------
  {header_text}
----------------------------------------------------------------------------------------------
"""

#FUNCTION TO DEFINE BUDGET LEVEL
def budget_level(level):
    return f"""Budget Level:    {level}"""

#FUNCTION TO DEFINE PRODUCT/SERVICE LEVEL AND SHOW PERCENTAGE
def product_service_level(category, department_expense):
    product_level = ""
    for item in sorted(set(category)): #set for assemble Category, stored by letter #Using for loop for category 
        product_level += f"""{item.capitalize()} ({round(department_expense.loc[department_expense['category'] == item, 'percentage'].values[0], 1)} %) :\n\n""" #.values[0] serier turn to Numpy array and retrieve the first element of the array
        filtered_rows = wedding_cost_data[wedding_cost_data['category'] == item]
        for product in filtered_rows.iterrows(): #_"not using the first value returned by iterrows()
            product_level += f"""{product[1]['item_name'].capitalize()}  {product[1]['subtotal']} \n\n"""
            product_level += """\n""" 
    return product_level

#FUNCTION TO DEPARTMENT LEVEL STRING
def department_level(department_expense):
    department_level = ""
    for i in range(10):
        department_level += f"""{department_expense["category"][i].upper()}:   {department_expense["department_expense"][i]}\n\n"""
        department_level += """\n""" 
    return department_level

#FUNCTION TO GENERATE TOTAL PRICE STRING
def total_price(total_price):
    return f"""Total Price :\t{total_price}"""

#SUM OF TOTAL DEPARTMENTAL EXPENSES
total_price_value = np.sum(department_expense['department_expense'])
#BUDGET LEVEL
level = wedding_cost_data.iloc[1, 0].capitalize()
#SEARCH CATEGORY
category = wedding_cost_data.iloc[:, 1]

#TEXT INTRO

import matplotlib.pyplot as plt
department_expense['percentage'] = department_expense['department_expense'] / department_expense['department_expense'].sum() * 100

#SIZE & PERCENTAGE
plt.figure(figsize=(15, 15))
colors = plt.cm.Paired(range(len(department_expense['category'])))

plt.title('Department Expense Distribution')

#CATEGORIES

plt.show()
